import React, { Component } from 'react';
import { View, Picker, Slider, Switch, Text} from 'react-native';
//import { Constants } from 'expo';

export default class App extends Component {
  constructor(props){
    super(props);
    
  }
  state = {
   pickerState1: {
    trueSwitchIsOn: true,
    selected1: 'key1',
    valueSlider: 1,
  },
  pickerState2: {
    trueSwitchIsOn: true,
    selected1: 'key2',
    valueSlider: 1,
  },
    trueSwitchIsOn: true,
    selected1: 'key1',
    valueSlider: 1,
  };
  
  
  
  fixa = {flexDirection: "row", height: 100};
  fixb = {flexDirection: "column", width: 100};
  
 initPickerState() {
   let pickerState2 = {
    trueSwitchIsOn: true,
    selected1: 'key1',
    valueSlider: 1,
  }
  
   let pickerState1 = {
    trueSwitchIsOn: true,
    selected1: 'key2',
    valueSlider: 1,
  }
  
  let pickerState3 = {
    trueSwitchIsOn: true,
    selected1: 'key3',
    valueSlider: 1,
  }
  
   this.pickerState.push("cell1" ,pickerState2);
   this.pickerState.push("cell2", pickerState1);
   this.pickerState.push("cell3", pickerState3);
   
 }
  
 turnDirection(trueSwitchIsOn) {
    if (trueSwitchIsOn) return {flexDirection: "row", height: 100};
    return {flex: 1, flexDirection: "column", width: 100};
  }
  
  render() {
    return (
      <View style={{flex: 1}}>
        <View style={{flex: 1, backgroundColor: "grey"}}>
                <View style={{flex: 1}}>
                    <Picker
                    selectedValue={this.state.selected1}
                    onValueChange={(value) => this.setState({pickerState1.selected1 : value})}>
                        <Picker.Item label="Cell 1" value="cell1"/>
                        <Picker.Item label="Cell2" value="cell2"/>
                        <Picker.Item label="Cell3" value="cell3"/>
                        
                    </Picker>
                    <Slider
                        {...this.props}
                        minimumValue={1}
                        maximumValue={7}
                        onValueChange={(value) => this.setState({valueSlider: value})}
                        />
                    <Switch
                    onValueChange={(value) => this.setState({trueSwitchIsOn: value})}
                    value={this.state.trueSwitchIsOn} />
                    <Text> {this.state.valueSlider} </Text>
                    <View style={this.turnDirection(this.state.trueSwitchIsOn)}>
                        <View style={{
                            flex: this.state.valueSlider,
                            backgroundColor: 'red',
                        }}/>
                        <View style={{flex: 1,  backgroundColor: 'green'}}/>
                        <View style={{flex: 1,  backgroundColor: 'blue'}}/>
                    </View>
                   
                </View>
            </View>
      </View>
    );
  }
}
